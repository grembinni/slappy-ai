      sounds_dir = ASSETS_DIR / "sounds"
      for wav in sorted(sounds_dir.glob("*.wav")):
          key = wav.stem.lower()
          self.sounds[key] = pygame.mixer.Sound(str(wav))

      GAME_OVER = auto()

    Define class Game with:
      __init__(self, screen: pygame.Surface, assets) — stores self.screen, self.assets; creates self.clock = pygame.time.Clock(); sets self.state = GameState.PLAYING; calls self.background = self._bake_background()

      _bake_background(self) -> pygame.Surface:
        Creates surf = pygame.Surface((SCREEN_W, SCREEN_H))
        surf.fill(SKY_COLOR)
        pygame.draw.rect(surf, CEILING_COLOR, (0, 0, SCREEN_W, CEILING_H))
        pygame.draw.rect(surf, GROUND_COLOR, (0, GROUND_Y, SCREEN_W, SCREEN_H - GROUND_Y))
        return surf

      update(self, dt: float) -> None:
        Phase 1: pass  (no gameplay entities yet; placeholder for later phases)

      draw(self) -> None:
        self.screen.blit(self.background, (0, 0))
        pygame.display.flip()

      run(self) -> None:
        running = True
        while running:
            dt = self.clock.tick(FPS) / 1000.0   ← FPS from settings, divide by float 1000.0
            for event in pygame.event.get():
                if event.type == pygame.QUIT:
                    running = False
                elif event.type == pygame.KEYDOWN:
                    if event.key == pygame.K_DELETE:
                        running = False
            self.update(dt)
            self.draw()
