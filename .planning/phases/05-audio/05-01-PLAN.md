      import pathlib
      import pygame

      _MUSIC_DIR = pathlib.Path(__file__).parent / "assets" / "sounds"


      class SoundManager:

          def __init__(self, sounds: dict) -> None:
              self._sounds = sounds
              self._muted: bool = False

          def play_laser(self) -> None:
              if self._muted:
                  return
              self._sounds['laser'].play()

          def play_death_cry(self) -> None:
              if self._muted:
                  return
              self._sounds['deathcry'].play()

          def play_explode(self) -> None:
              if self._muted:
                  return
              self._sounds['explode'].play()

          def play_playing_music(self) -> None:
              pygame.mixer.music.stop()
              pygame.mixer.music.load(str(_MUSIC_DIR / "passport.wav"))
              pygame.mixer.music.play(-1)

          def play_splash_music(self) -> None:
              pygame.mixer.music.stop()
              pygame.mixer.music.load(str(_MUSIC_DIR / "canyon.wav"))
              pygame.mixer.music.play(-1)

          def stop_music(self) -> None:
              pygame.mixer.music.stop()

          def toggle_mute(self) -> None:
              self._muted = not self._muted
              if self._muted:
                  pygame.mixer.pause()
                  pygame.mixer.music.pause()
              else:
                  pygame.mixer.unpause()
                  pygame.mixer.music.unpause()

    CORRECTNESS NOTES:
    - _MUSIC_DIR is module-level; pathlib.Path(__file__).parent resolves to the project root where sound.py lives — same pattern as assets.py's ASSETS_DIR
    - SFX use self._sounds[key].play() — Sound objects are already loaded by AssetCache; no load needed
    - Music methods call stop() first (D-06) to cleanly switch tracks and prevent overlaps
    - Mute uses pause/unpause so mid-play sounds and music resume from their paused position (D-10, D-11)
    - No mute guard on music methods: play_playing_music() starts music regardless of mute state; toggle_mute() handles pause after the fact. This is correct — the game always starts unmuted.
    - intro.wav exists in assets/sounds/ but has no AUD requirement — do NOT reference it
