    CLASS-LEVEL SPRITE TABLES (define at class scope before __init__):

      _CONTROLS: dict mapping character name to {DIR: key_constant}
        "superman": {DIR_UP: pygame.K_UP, DIR_DOWN: pygame.K_DOWN, DIR_LEFT: pygame.K_LEFT, DIR_RIGHT: pygame.K_RIGHT}
        "goblin":   {DIR_UP: pygame.K_e,  DIR_DOWN: pygame.K_d,    DIR_LEFT: pygame.K_s,    DIR_RIGHT: pygame.K_f}

        "superman": {
            DIR_UP:    ("sup1",    "sup2"),
            DIR_DOWN:  ("sdown1",  "sdown2"),
            DIR_LEFT:  ("sleft1",  "sleft2"),
            DIR_RIGHT: ("sright1", "sright2"),
            DIR_IDLE:  ("superman","superman"),
        }

            DIR_UP:    ("gup1",  "gup2"),
            DIR_DOWN:  ("gdown1","gdown2"),
            DIR_LEFT:  ("gleft1","gleft2"),
            DIR_RIGHT: ("gright1","gright2"),
            DIR_IDLE:  ("gevil", "gevil"),
        }

      _GROUND_WALK_CYCLE = [0, 1, 0, 2]   # 4-step index list — D-10: cycle 1->2->1->3 (0-indexed)

      _GROUND_WALK_SPRITES: dict mapping character name to list of 3 sprite keys (0-indexed)
        "superman": ["ckent1", "ckent2", "ckent3"]
        "goblin":   ["gevil1", "gevil2", "gevil3"]

      _GROUND_IDLE_SPRITE: dict mapping character name to ground-idle sprite key
        "superman": "ckent"
        "goblin":   "gevil"

    __init__(self, character: str, assets):
      self.character = character
      self.assets = assets
      self.state = CharState.ALIVE
      Starting positions (per Claude's Discretion in CONTEXT.md):
        if character == "superman":
            self.x = SCREEN_W * 0.75
            self.facing = DIR_LEFT
        else:
            self.x = SCREEN_W * 0.25
            self.facing = DIR_RIGHT
      self.y = float(SCREEN_H // 2)
      self.x = float(self.x)    # ensure float
      Animation state:
        self._anim_timer = 0.0
        self._anim_frame = 0
        self._ground_walk_step = 0
      Cached draw state (updated each update() call):
        self._direction = DIR_IDLE
        self._on_ground = False
        self._moving_h = False

    rect property:
      @property
      def rect(self) -> pygame.Rect:
          return pygame.Rect(int(self.x), int(self.y), SPRITE_SIZE, SPRITE_SIZE)
      # Used by Phase 3 beam collision via pygame.Rect.clipline()

    update(self, dt: float, keys) -> None:
      Only execute body if self.state == CharState.ALIVE.

      Step 1 — determine on_ground:
        on_ground = self.y >= GROUND_Y

      Step 2 — compute dx, dy, direction:
        dx = 0.0
        dy = 0.0
        direction = DIR_IDLE
        ctrl = self._CONTROLS[self.character]

        Horizontal (use elif — only one direction applies per frame):
          if keys[ctrl[DIR_LEFT]]:
              dx = -PLAYER_SPEED_H * dt
              direction = DIR_LEFT
              self.facing = DIR_LEFT
          elif keys[ctrl[DIR_RIGHT]]:
              dx = PLAYER_SPEED_H * dt
              direction = DIR_RIGHT
              self.facing = DIR_RIGHT

        Vertical (only updates direction if no horizontal key was held):
          if keys[ctrl[DIR_UP]]:
              dy = -PLAYER_SPEED_UP * dt
              if direction == DIR_IDLE:
                  direction = DIR_UP
                  self.facing = DIR_UP
          elif keys[ctrl[DIR_DOWN]] and not on_ground:
              dy = PLAYER_SPEED_DOWN * dt
              if direction == DIR_IDLE:
                  direction = DIR_DOWN
                  self.facing = DIR_DOWN
        # NOTE: down key is suppressed on ground (D-11 — visual floor only, not a state change)

      Step 3 — apply movement:
        self.x += dx
        self.y += dy

      Step 4 — horizontal wrap:
        # Characters wrap at screen edges (MOV-04)
        if self.x < -SPRITE_SIZE:
            self.x = SCREEN_W
        elif self.x > SCREEN_W:
            self.x = -SPRITE_SIZE

      Step 5 — vertical clamp:
        # Ceiling (MOV-06): clamp to CEILING_H
        if self.y < CEILING_H:
            self.y = CEILING_H
        # Ground (MOV-05): clamp to GROUND_Y
        if self.y > GROUND_Y:
            self.y = GROUND_Y

      Step 6 — re-compute on_ground after clamp:
        on_ground = self.y >= GROUND_Y

      Step 7 — animation timer:
        self._anim_timer += dt
        if self._anim_timer >= ANIM_INTERVAL:
            self._anim_timer -= ANIM_INTERVAL       # subtract, don't reset to 0 (preserves sub-interval remainder)
            self._anim_frame = 1 - self._anim_frame  # toggle 0<->1
            if on_ground and dx != 0:
                self._ground_walk_step = (self._ground_walk_step + 1) % 4

      Step 8 — cache draw state:
        self._direction = direction
        self._on_ground = on_ground
        self._moving_h = (dx != 0)

    draw(self, screen: pygame.Surface) -> None:
      key = self._get_sprite_key()
      screen.blit(self.assets.sprites[key], (int(self.x), int(self.y)))

    _get_sprite_key(self) -> str:
      if self._on_ground:
          if self._moving_h:
              idx = self._GROUND_WALK_CYCLE[self._ground_walk_step]
              return self._GROUND_WALK_SPRITES[self.character][idx]
          else:
              return self._GROUND_IDLE_SPRITE[self.character]
      else:
          frames = self._AIR_SPRITES[self.character][self._direction]
          return frames[self._anim_frame]

    PHASE 3 NOTE (add as comment above the facing attribute assignment in __init__):
      # self.facing: last non-idle direction — used by Phase 3 beam firing.
      # Phase 3 collision check MUST use: if beam.direction in (LEFT, RIGHT):
      # NEVER write: beam.direction == LEFT or RIGHT  (VB6 Or-bug — always True for truthy RIGHT)
