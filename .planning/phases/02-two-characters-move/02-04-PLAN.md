    MODULE HEADER — environment setup (must be at the very top, before any pygame import):
      import os
      os.environ.setdefault("SDL_VIDEODRIVER", "dummy")
      os.environ.setdefault("SDL_AUDIODRIVER", "dummy")

    IMPORTS:
      import pytest
      import pygame
      from unittest.mock import MagicMock
      from collections import defaultdict
      import settings
      from player import Player, CharState
      from settings import (
          SCREEN_W, SCREEN_H, SPRITE_SIZE, CEILING_H, GROUND_Y,
          PLAYER_SPEED_H, PLAYER_SPEED_UP, PLAYER_SPEED_DOWN, ANIM_INTERVAL,
          DIR_UP, DIR_DOWN, DIR_LEFT, DIR_RIGHT, DIR_IDLE,
      )

    FIXTURES:
      @pytest.fixture(autouse=True)
      def pygame_init():
          pygame.init()
          yield
          pygame.quit()

      def make_assets():
          """MagicMock AssetCache — all sprite keys return a 128x128 Surface."""
          assets = MagicMock()
          assets.sprites = defaultdict(lambda: pygame.Surface((SPRITE_SIZE, SPRITE_SIZE)))
          return assets

      def make_keys(**pressed):
          """Simulate pygame.key.get_pressed() — returns defaultdict(bool) with specified keys True."""
          d = defaultdict(bool)
          d.update(pressed)
          return d

    TESTS — write exactly these 20 test functions in this order:

    1. test_charstate_enum:
       from enum import Enum
       assert issubclass(CharState, Enum)
       for name in ("ALIVE", "CRASHING", "DEAD", "POSING", "RESPAWNING"):
           assert hasattr(CharState, name), f"CharState missing {name}"
       # MOV-07

    2. test_player_initial_state_superman:
       p = Player("superman", make_assets())
       assert p.state == CharState.ALIVE
       assert p.x == pytest.approx(SCREEN_W * 0.75)
       assert p.y == SCREEN_H // 2
       # MOV-07

    3. test_player_initial_state_goblin:
       p = Player("goblin", make_assets())
       assert p.state == CharState.ALIVE
       assert p.x == pytest.approx(SCREEN_W * 0.25)
       assert p.y == SCREEN_H // 2
       # MOV-07

    4. test_superman_moves_right:
       p = Player("superman", make_assets())
       orig_x = p.x
       p.update(0.1, make_keys(**{pygame.K_RIGHT: True}))
       assert p.x == pytest.approx(orig_x + PLAYER_SPEED_H * 0.1)
       # MOV-01, MOV-03

    5. test_superman_moves_left:
       p = Player("superman", make_assets())
       orig_x = p.x
       p.update(0.1, make_keys(**{pygame.K_LEFT: True}))
       assert p.x == pytest.approx(orig_x - PLAYER_SPEED_H * 0.1)
       # MOV-01, MOV-03

    6. test_superman_moves_up:
       p = Player("superman", make_assets())
       orig_y = p.y
       p.update(0.1, make_keys(**{pygame.K_UP: True}))
       assert p.y == pytest.approx(orig_y - PLAYER_SPEED_UP * 0.1)
       # MOV-01, MOV-03

    7. test_superman_moves_down:
       # Character must be in air (not at GROUND_Y) for down key to work
       p = Player("superman", make_assets())
       p.y = float(SCREEN_H // 2)   # midscreen — in air
       orig_y = p.y
       p.update(0.1, make_keys(**{pygame.K_DOWN: True}))
       assert p.y > orig_y
       assert p.y == pytest.approx(orig_y + PLAYER_SPEED_DOWN * 0.1)
       # MOV-01, MOV-03

    8. test_goblin_moves_right:
       p = Player("goblin", make_assets())
       orig_x = p.x
       p.update(0.1, make_keys(**{pygame.K_f: True}))
       assert p.x == pytest.approx(orig_x + PLAYER_SPEED_H * 0.1)
       # MOV-02, MOV-03

    9. test_ceiling_clamp:
       p = Player("superman", make_assets())
       p.y = float(CEILING_H + 1)   # just above ceiling — pressing up should be clamped
       p.update(0.5, make_keys(**{pygame.K_UP: True}))
       assert p.y >= CEILING_H, f"y={p.y} should be >= CEILING_H={CEILING_H}"
       # MOV-06

    10. test_ground_clamp:
        p = Player("superman", make_assets())
        p.y = float(SCREEN_H)   # below ground boundary
        p.update(0.016, make_keys())
        assert p.y <= GROUND_Y, f"y={p.y} should be <= GROUND_Y={GROUND_Y}"
        # MOV-05

    11. test_horizontal_wrap_right:
        p = Player("superman", make_assets())
        p.x = float(SCREEN_W + 10)
        p.update(0.016, make_keys(**{pygame.K_RIGHT: True}))
        assert p.x < SCREEN_W, f"Expected wrap, got x={p.x}"
        # MOV-04

    12. test_horizontal_wrap_left:
        p = Player("superman", make_assets())
        p.x = float(-SPRITE_SIZE - 10)
        p.update(0.016, make_keys(**{pygame.K_LEFT: True}))
        assert p.x >= 0, f"Expected wrap to right side, got x={p.x}"
        # MOV-04

    13. test_rect_property:
        p = Player("superman", make_assets())
        r = p.rect
        assert isinstance(r, pygame.Rect)
        assert r.x == int(p.x)
        assert r.y == int(p.y)
        assert r.width == SPRITE_SIZE
        assert r.height == SPRITE_SIZE
        # Phase 3 collision prerequisite

    14. test_animation_frame_toggles:
        p = Player("superman", make_assets())
        p._anim_frame = 0
        p._anim_timer = ANIM_INTERVAL - 0.001
        p.update(0.002, make_keys())   # push timer over threshold
        assert p._anim_frame == 1, "Expected frame to toggle from 0 to 1"
        # MOV-08

    15. test_get_sprite_key_idle_air:
        # Fresh superman is in air at SCREEN_H//2 (well above GROUND_Y=750)
        p = Player("superman", make_assets())
        p._direction = DIR_IDLE
        p._on_ground = False
        p._anim_frame = 0
        key = p._get_sprite_key()
        assert key == "superman", f"Expected 'superman', got '{key}'"
        # MOV-09

    16. test_get_sprite_key_ground_idle:
        p = Player("superman", make_assets())
        p._on_ground = True
        p._moving_h = False
        key = p._get_sprite_key()
        assert key == "ckent", f"Expected 'ckent', got '{key}'"
        # MOV-09

    17. test_get_sprite_key_ground_walk:
        p = Player("superman", make_assets())
        p._on_ground = True
        p._moving_h = True
        p._ground_walk_step = 0   # GROUND_WALK_CYCLE[0] = 0 -> "ckent1"
        key = p._get_sprite_key()
        assert key == "ckent1", f"Expected 'ckent1', got '{key}'"
        p._ground_walk_step = 1   # GROUND_WALK_CYCLE[1] = 1 -> "ckent2"
        key = p._get_sprite_key()
        assert key == "ckent2", f"Expected 'ckent2', got '{key}'"
        p._ground_walk_step = 3   # GROUND_WALK_CYCLE[3] = 2 -> "ckent3"
        key = p._get_sprite_key()
        assert key == "ckent3", f"Expected 'ckent3', got '{key}'"
        # MOV-08, MOV-09, D-10

    18. test_no_downward_movement_on_ground:
        p = Player("superman", make_assets())
        p.y = float(GROUND_Y)   # exactly at ground
        p.update(0.1, make_keys(**{pygame.K_DOWN: True}))
        assert p.y == GROUND_Y, f"Down key should be suppressed on ground, got y={p.y}"
        # MOV-05, D-11

    19. test_dt_scaling:
        # Movement distance should be proportional to dt
        p1 = Player("superman", make_assets())
        p1.update(0.1, make_keys(**{pygame.K_RIGHT: True}))
        dist1 = p1.x - (SCREEN_W * 0.75)

        p2 = Player("superman", make_assets())
        p2.update(0.05, make_keys(**{pygame.K_RIGHT: True}))
        dist2 = p2.x - (SCREEN_W * 0.75)

        assert dist1 == pytest.approx(dist2 * 2, rel=1e-5), (
            f"dt scaling broken: 0.1s gave {dist1}px, 0.05s gave {dist2}px (expected 2x ratio)"
        )
        # MOV-03, CLAUDE.md dt constraint

    20. test_both_players_move_independently:
        sup = Player("superman", make_assets())
        gob = Player("goblin", make_assets())
        sup_orig_x = sup.x
        gob_orig_x = gob.x
        # Press both right keys simultaneously
        keys = make_keys(**{pygame.K_RIGHT: True, pygame.K_f: True})
        sup.update(0.1, keys)
        gob.update(0.1, keys)
        assert sup.x > sup_orig_x, "Superman should have moved right"
        assert gob.x > gob_orig_x, "Goblin should have moved right"
        # MOV-01, MOV-02 — shared keyboard, independent movement
